consul命令

使用consul agent -h 可以查看帮忙命令

一些常用的参数有：

-bind 指定consul所在ip地址

-client 表明哪些机器可以访问consul，0.0.0.0不受限

-config-dir 所有主动注册服务的描述信息

-data-dir 存储所有注册过来的srv机器的详细信息

-dev 开发者模式

-http-port consul自带一个web访问的默认端口号：8500

-node 服务发现的名字

-rejoin 重新加入集群

-server 切换是否作为server模式；不加则作为client模式

-ui 使用内置UI静态web服务

consul 使用：

带有一些参数启动

#在终端键入
consul agent -server -bootstrap-expect 1 -data-dir="D:\Program Files\consul_1.11.4_windows_amd64\temp" -node=n1 -bind=192.168.0.54 -ui -rejoin -config-dir="D:\Program Files\consul_1.11.4_windows_amd64\conf" -client=0.0.0.0

consul结合grpc：

流程：

1、创建proto文件，指定rpc服务
2、启动consul服务发现
3、启动server
		3.1 获取consul对象
		3.2 使用consul对象，将server信息，注册给consul
		3.3 启动服务
4、启动Client
		4.1 获取consul对象
		4.2 使用consul对象，从consul上获取健康的服务
		4.3 再访问服务（grpc远程调用）

示例：

1、定义person.proto

syntax="proto3";
package pb;
option go_package="./";
message Person{
  int32 age=1;
  string name=2;
}

service Hello{
  rpc SayHello(Person) returns (Person);
}

执行命令来生成pb.go文件protoc --go_out=plugins=grpc:./ *.proto

2、consul服务发现

server.go

package main

import (
	pb "StudyGrpcAndConsul/pb"
	"context"
	"fmt"
	"github.com/hashicorp/consul/api"
	"google.golang.org/grpc"
	"net"
)

type Children struct{}

func (this *Children) SayHello(ctx context.Context, p *pb.Person) (*pb.Person, error) {
	p.Name = "hello" + p.Name
	return p, nil
}

func main() {

	//初始化consul配置
	consulConfig := api.DefaultConfig()
	//获取consul操作对象
	consulClient, _ := api.NewClient(consulConfig)
	//注册服务，服务的常规配置
	registerService := api.AgentServiceRegistration{
		ID:      "1",
		Tags:    []string{"testHello"},
		Name:    "HelloService",
		Port:    9090,
		Address: "127.0.0.1",
		Check: &api.AgentServiceCheck{
			TCP:      "127.0.0.1:9090",
			Timeout:  "5s",
			Interval: "5s",
		},
	}
	//注册服务到consul上
	consulClient.Agent().ServiceRegister(&registerService)

	///////////////以下为grpc远程调用/////////////////
	//1 初始化grpc对象
	grpcServer := grpc.NewServer()
	//2 注册服务
	pb.RegisterHelloServer(grpcServer, new(Children))
	//3 设置监听
	lis, err := net.Listen("tcp", "127.0.0.1:9090")
	if err != nil {
		fmt.Println("监听失败：", err)
		return
	}
	defer lis.Close()
	fmt.Println("启动监听")
	//4 启动服务
	grpcServer.Serve(lis)

}

client.go

package main

import (
	pb "StudyGrpcAndConsul/pb"
	"context"
	"fmt"
	"github.com/hashicorp/consul/api"
	"google.golang.org/grpc"
	"strconv"
)

func main() {
	//1 初始化consul配置
	consulConfig := api.DefaultConfig()
	//2 创建consul对象，（可以重新指定consul的ip、port等属性，也可以使用默认）
	consulClient, err := api.NewClient(consulConfig)
	if err != nil {
		fmt.Println("创建consul对象失败：", err)
		return
	}
	//3 服务发现，从consul上获取健康的服务
	services, _, err := consulClient.Health().Service("HelloService", "testHello", true, nil)
	if err != nil {
		fmt.Println("获取服务发现失败：err", err)
		return
	}
	addr := services[0].Service.Address + ":" + strconv.Itoa(services[0].Service.Port)
	fmt.Println()
	///////////以下为grpc服务远程调用///////////////
	//1 连接grpc客户端
	//grpcConn, err := grpc.Dial("127.0.0.1:9090", grpc.WithInsecure())
	//修改为从服务发现中得到的addr
	grpcConn, err := grpc.Dial(addr, grpc.WithInsecure())
	if err != nil {
		fmt.Println("连接grpc出错：", err)
		return
	}
	defer grpcConn.Close()
	//2 初始化grpc客户端
	grpcClient := pb.NewHelloClient(grpcConn)
	//3 调用远程服务
	var per pb.Person
	per.Name = "abin"
	per.Age = 18
	p, err := grpcClient.SayHello(context.TODO(), &per)
	if err != nil {
		fmt.Println("grpc调用出错:", err)
		return
	}
	fmt.Println(p)
}

go-micro

1、编写proto文件:hello.proto

package pb;

option  go_package="./;pb";

message Request{
  string name=1;
}

message Response{
  string greeting=1;
}

service Greeter{
  rpc Hello(Request) returns (Response);
}

使用命令：protoc --proto_path=. --micro_out=. --go_out=. *.proto 来生成micro.go和pb.go

2、实现服务端

package main

import (
	pb "StudyGoMicro/pb"
	"context"
	"fmt"
	micro "github.com/micro/go-micro/v2"
)

type Greeter struct{}

func (g *Greeter) Hello(ctx context.Context, req *pb.Request, rsp *pb.Response) error {
	rsp.Greeting = "hello" + req.Name
	return nil
}

func main() {
	//1、创建一个新的服务
	service := micro.NewService(
		micro.Name("greeter"),
	)
	//2、初始化服务
	service.Init()
	//3、注册Handler
	pb.RegisterGreeterHandler(service.Server(), new(Greeter))
	//4、运行服务
	err := service.Run()
	if err != nil {
		fmt.Println(err)
	}
}

3、实现客户端

import (
	"StudyGoMicro/pb"

Server.go的代码

1、要初始化consul服务发现consulReg:=consul.NewRegistry()

2、添加服务micro.Registry(consulReg)

注意consul包引用地址；

package main

import (
	pb "StudyGoMicro/pb"
	"context"
	"fmt"
	micro "github.com/micro/go-micro/v2"
	"github.com/micro/go-micro/v2/registry"
	"github.com/micro/go-plugins/registry/consul/v2"
)

type Greeter struct{}

func (g *Greeter) Hello(ctx context.Context, req *pb.Request, rsp *pb.Response) error {
	rsp.Greeting = "hello" + req.Name
	return nil
}

func main() {
	// 0 初始化consul
	consulReg := consul.NewRegistry(func(options *registry.Options) {
		options.Addrs = []string{
			"127.0.0.1:8500",
		}
	})
	//1、创建一个新的服务
	service := micro.NewService(
		micro.Name("greeter"),
		micro.Registry(consulReg),
	)
	//2、初始化服务
	service.Init()
	//3、注册Handler
	pb.RegisterGreeterHandler(service.Server(), new(Greeter))
	//4、运行服务
	err := service.Run()
	if err != nil {
		fmt.Println(err)
	}
}

package main
