使用

准备proto文件

syntax="proto3";

package pb;

option go_package="/proto";

message Teacher{
  int32 age=1;
  string name=2;

}

service SayName{
  rpc SayHello(Teacher) returns(Teacher);
}

执行命令来生成pb.go文件protoc --go_out=plugins=grpc:./ *.proto

调用：

服务端：

1、初始化一个grpc对象

2、注册服务

3、设置监听、指定IP、port

4、启动服务

package main

import (
	"context"
	"fmt"
	"google.golang.org/grpc"
	pb "StudyGRPC/pb/proto"
	"net"
)

type Children struct {

}

func (this *Children) SayHello(ctx context.Context,t *pb.Teacher)(*pb.Teacher,error){
	t.Name+="is studying"
	return t,nil
}

func main(){
	//1、初始化一个grpc对象
	grpcServer:=grpc.NewServer()
	//2、注册服务
	pb.RegisterSayNameServer(grpcServer,new(Children))
	//3、设置监听
	listen, err := net.Listen("tcp", "127.0.0.1:8080")
	if err!=nil{
		fmt.Println("监听失败：",err)
		return
	}
	defer listen.Close()
	fmt.Println("启动监听成功")
	//4、启动服务
	grpcServer.Serve(listen)

}

客户端：

1、连接grpc服务

2、初始化grpc客户端

3、调用远程服务

package main

import (
	pb "StudyGRPC/pb/proto"
	"context"
	"fmt"
	"google.golang.org/grpc"
)

func main(){
	//1、连接grpc服务
	grpcConn, err := grpc.Dial("127.0.0.1:8080", grpc.WithInsecure())
	if err!=nil{
		fmt.Println("连接grpc出错：",err)
		return
	}
	defer grpcConn.Close()
	//2、初始化grpc客户端
	grpcClient := pb.NewSayNameClient(grpcConn)
	//3、调用grpc服务
	var teacher pb.Teacher
	teacher.Name="abin"
	teacher.Age=18
	t, err := grpcClient.SayHello(context.TODO(), &teacher)
	if err!=nil{
		fmt.Println("grpc 调用出错:",err)
		return
	}
	fmt.Println(t)
}
