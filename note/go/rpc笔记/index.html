RPC笔记

作用：

RPC：remote procedure call protocol远程调用协议；基于TCP协议，在网络中，实现像调用本机的函数一样，调用其他机子的函数。

过程：

在服务端：

1、注册服务，并绑定方法

rpc.RegisterName("服务名",回调对象)

回调对象中的方法需要满足以下条件：

  • 1、方法是导出的：也就是首字母要大写
  • 2、方法有两个参数，都是导出类型或内建类型
  • 3、方法的第二个参数是指针
  • 4、方法只有一个error接口类型的返回值

例如：

type Hello struct{}

func (h *Hello)HelloWorld(name string,resp *string) error{ *resp =“你好，"+name return nil }

2、启动监听

listener,err:=net.Listen()

3、建立连接

conn,err:=listener.Accept()

4、将连接绑定到rpc服务

rpc.ServeConn(conn)

在客户端：

1、用rpc连接服务器

conn,err:=rpc.Dial()

2、调用远程函数

conn.Call("服务名.方法名",传入参数，传出参数)

example:

Server.go

package main

import (
	"net/rpc"
	"net"
	"fmt"
)

type Hello struct{}

func (h *Hello)HelloWorld(name string,resp *string) error{
	*resp ="你好，"+name
	return nil
}


func main(){
	//1、注册rpc服务，并绑定方法
	err:=rpc.RegisterName("hello",new(Hello))
	if err!=nil{
		fmt.Println("rpc注册失败:",err)
		return
	}
	//2、创建监听
	listener,err:=net.Listen("tcp","127.0.0.1:8080")
	if err!=nil{
		fmt.Println("监听失败:",err)
		return
	}
	defer listener.Close()
	fmt.Println("启动监听成功")
	//3、建立连接
	conn,err:=listener.Accept()
	if err!=nil{
		fmt.Println("建立连接失败:",err)
		return
	}
	defer conn.Close()
	//4、将连接绑定到服务
	rpc.ServeConn(conn)
}

client.go

package main

import(
	"net/rpc"
	"fmt"
)

func main(){
	//1、创建连接
	conn,err:=rpc.Dial("tcp","127.0.0.1:8080")
	if err!=nil{
		fmt.Println("连接失败:",err)
		return
	}
	defer conn.Close()
	//2、远程调用
	var reply string
	err=conn.Call("hello.HelloWorld","abin",&reply)
	if err!=nil{
		fmt.Println("远程调用失败:",err)
		return
	}
	fmt.Println(reply)
}

封装

创建一个封装文件，比如DesignRpc.go

package main

import(
	"net/rpc"
)

/*封装服务端*/

type MyInterface interface{
	HelloWorld(string, *string) error
}
//注册rpc服务，并绑定方法
func RegisterServer(i MyInterface){
	rpc.RegisterName("hello",i)
}

/*封装客户端*/
//定义客户端结构体
type MyClient struct{
	c *rpc.Client
}
//创建连接
func InitClient(addr string) MyClient{
	conn,_:=rpc.Dial("tcp",addr)
	return MyClient{c:conn}
}
//远程调用
func (this *MyClient) HelloWorld(a string,b *string) error{
	return this.c.Call("hello.HelloWorld",a,b)
}

server中就可以使用封装好的方法

package main

import (
	"net/rpc"
	"net"
	"fmt"
)

type Hello struct{

}

func (this *Hello) HelloWorld(a string, b *string) error{
	*b = a +" 你好！"
	return nil
}

func main(){
	RegisterServer(new(Hello))
	lis,err:=net.Listen("tcp","127.0.0.1:8080")
	if err!=nil{
		fmt.Println("listen err:",err)
		return
	}
	defer lis.Close()
	fmt.Println("listen start")
	conn,err:=lis.Accept()
	if err!=nil{
		fmt.Println("conn err:",err)
		return
	}
	defer conn.Close()
	rpc.ServeConn(conn)
}
