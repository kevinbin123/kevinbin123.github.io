Golang小案例-获取成语并查询

前言：

本案例实现的功能：

1、通过[万维易源]接口获取成语

2、解析获取到的内容，并通过成语title获取具体是成语释义

3、解析释义，并持久化到本地文件

4、根据命令行参数来实现精确查询或者模糊查询

本案例用到的知识为：

1、通过http.Get获取接口内容

2、通过json.Unmarsal解析内容

3、通过json的Encode方法持久化数据

4、通过flag实现命令行参数的解析

准备工作

定义成语的结构体

type Idiom struct {
	Title      string `json:"title"`
	Spell      string `json:"spell"`
	Smiple     string `json:"simple"`
	Derivation string `json:"derivation"`
	Content    string `json:"content"`
}

使用一个map来存放所有的成语

var Idioms = make(map[string]Idiom)

定义一个本地路径用来持久化数据

var DBpath = "成语大全.json"

定义用来接收命令行参数的变量

var StringVar string
var KeyVar string

获取列表数据

从接口获取列表数据

func GetJson(url string) (bytes []byte, err error) {
	resp, err := http.Get(url)
	if err != nil {
		fmt.Println("获取网址错误:", err)
		return
	}
	defer resp.Body.Close()
	bytes, err = ioutil.ReadAll(resp.Body)
	if err != nil {
		fmt.Println("获取网址内容错误：", err)
		return
	}
	return
}

解析列表数据

将获取到的列表数据解析到map中

func ParseList(bytes []byte) {
	var tempData map[string]interface{}
	json.Unmarshal(bytes, &tempData)
	i := tempData["showapi_res_body"].(map[string]interface{})["data"].([]interface{})
	for _, v := range i {
		idiom := Idiom{}
		tempTitle := v.(map[string]interface{})["title"].(string)
		idiom.Title = tempTitle
		Idioms[tempTitle] = idiom
	}
}

根据列表获取到详细信息并解析

根据列表数据获取到的内容，作为关键字，发起查询具体释义的接口，获取到每个成语具体的释义，然后再解析到map中

func GetDetailbyListAndParse() {
	for k, _ := range Idioms {
		bytes, err := GetJson("https://route.showapi.com/1196-2?showapi_appid=919595&showapi_sign=1effaf63b2bf44a8bdb60ae194b4f83b&keyword=" + k)
		if err != nil {
			fmt.Println("获取解释失败:", err)
			continue
		}
		//fmt.Println(string(bytes))
		var tempData map[string]interface{}
		json.Unmarshal(bytes, &tempData)
		m := tempData["showapi_res_body"].(map[string]interface{})["data"].(map[string]interface{})
		fmt.Println(m)
		fmt.Println(m["spell"].(string))
		idiom := Idiom{}
		idiom.Spell = m["spell"].(string)
		idiom.Derivation = m["derivation"].(string)
		idiom.Content = m["content"].(string)
		idiom.Title = m["title"].(string)
		Idioms[k] = idiom
		time.Sleep(1 * time.Second)

	}
}

持久化数据

将map中的数据保存到本地文件

func PersistentData() {
	file, err := os.OpenFile(DBpath, os.O_WRONLY|os.O_CREATE|os.O_TRUNC, 0666)
	if err != nil {
		fmt.Println("创建本地文件失败：", err)
		return
	}
	defer file.Close()
	encoder := json.NewEncoder(file)
	err = encoder.Encode(Idioms)
	if err != nil {
		fmt.Println("写入数据到json失败:", err)
		return
	} else {
		fmt.Println("写入数据成功")
	}
}

从文件读取到内存

把文件读取到内存中

func ReadFromJson(path string) {
	file, err := os.OpenFile(path, os.O_RDONLY, 0666)
	if err != nil {
		fmt.Println("读取文件失败：", err)
		return
	}
	defer file.Close()
	decoder := json.NewDecoder(file)
	err = decoder.Decode(&Idioms)
	if err != nil {
		fmt.Println("加载数据失败：", err)
	} else {

		fmt.Println("从文件中成功加载数据！")
	}
}

判断是否有数据

根据本地是否有指定的文件，如果已经有了，则认为已经获取过数据，执行从文件中读取的动作；如果没有此文件，则认为没有抓取过，执行重新抓取的动作。

func init() {
	fileInfo, err := os.Stat(DBpath)
	if err == nil && fileInfo != nil {
		fmt.Println("文件已存在，不再抓取，直接读取")
		ReadFromJson(DBpath)
	} else {
		fmt.Println("文件不存在，开始抓取")
		//1、获取列表内容
		bytes, _ := GetJson("https://route.showapi.com/1196-1?showapi_appid=919595&showapi_sign=1effaf63b2bf44a8bdb60ae194b4f83b&keyword=色&page=1&rows=10")
		//2、解析内容
		//2.1 解析列表内容
		ParseList(bytes)
		//fmt.Println(Idioms)
		// 2.2 根据列表内容获取详细内容
		// 2.3 解析详细内容
		time.Sleep(1 * time.Second)
		GetDetailbyListAndParse()

		//3、持久化到本地
		PersistentData()
	}

}

命令行参数解析

根据解析到的命令行看是模糊查找，还是精确查找

func main() {
	flag.StringVar(&StringVar, "cmd", "ambiguous", "ambiguous模糊查找；accurate精确查找")
	flag.StringVar(&KeyVar, "key", "关键字", "输入关键字")
	flag.Parse()
	//fmt.Println("StringVar,KeyVar", StringVar, KeyVar)
	if StringVar == "ambiguous" {
		fmt.Println("选择的是模糊查找，关键字为：", KeyVar)
		for title, v := range Idioms {
			if strings.Contains(title, KeyVar) {
				fmt.Println(v)
			}
		}
	} else if StringVar == "accurate" {
		fmt.Println("选择的是精确查找，关键字为：", KeyVar)
		if idiom, ok := Idioms[KeyVar]; !ok {
			fmt.Println("不存在这个成语：", KeyVar)
		} else {
			fmt.Println(idiom)
		}
	} else {
		fmt.Println("非法命令")
	}
}
