基本使用

设置string类型的键值对并读取

可以包含任何数据，一个键最大能存储512M

127.0.0.1:6379> set mykey "myvalue"
OK
127.0.0.1:6379> get mykey
"myvalue"

删除key

127.0.0.1:6379> del mykey
(integer) 1

设置hash类型的键值对并读取

适合存储对象，存储、读取、修改用户属性

127.0.0.1:6379> hmset mykey field1 "hello" field2 "redis"
OK
127.0.0.1:6379> hget mykey field2
"redis"
127.0.0.1:6379> hget mykey field1
"hello"

设置list类型的并读取

双向链表，可以用来实现消息队列

127.0.0.1:6379> lpush mykey redis
(integer) 1
127.0.0.1:6379> lpush mykey mongodb
(integer) 2
127.0.0.1:6379> lpush mykey sqlite
(integer) 3
127.0.0.1:6379> rpush mykey mysql
(integer) 4
127.0.0.1:6379> lrange mykey 0 100
1) "sqlite"
2) "mongodb"
3) "redis"
4) "mysql"

设置set集合并读取

添加、删除、查找复杂度都是O(1)，用来实现共同好友、好友推荐、统计访问网站的独立IP

127.0.0.1:6379> sadd mykey redis
(integer) 1
127.0.0.1:6379> sadd mykey mongodb
(integer) 1
127.0.0.1:6379> sadd mykey sqlite
(integer) 1
127.0.0.1:6379> sadd mykey sqlite
(integer) 0
127.0.0.1:6379> smembers mykey
1) "sqlite"
2) "mongodb"
3) "redis"

设置zset集合（有序集合）并读取

有序的集合，适合实现排行榜、带权重的消息队列等场景

基本使用

Keys()：根据正则获取keys,正则*表示获取所有的key

Type()：获取key对应的类型

Exists()：检测缓存项是否存在

Expire()，ExpireAt()：设置有效期

TTL(),PTTL()：获取有效期

DBSize()：查看当前数据库key的数量

Del()：删除缓存项

FlushDB()：清空当前数据

FlushAll()：清空所有数据库

package main

import (
	"context"
	"fmt"
	"github.com/go-redis/redis/v8"
	"time"
)

var ctx=context.Background()

func main(){
	rdb:=redis.NewClient(&redis.Options{
		Addr:"192.168.145.151:6379",
		Password:"",
		DB:0,
	})

	//使用Keys,匹配模式用*，查看所有的key
	result, err:= rdb.Keys(ctx, "*").Result()
	if err!=nil{
		panic(err)
	}
	fmt.Println(result)

	//使用Type,查看某个key是什么类型
	result2,err:=rdb.Type(ctx,"lan_rank").Result()
	if err!=nil{
		panic(err)
	}
	fmt.Println(result2)

	//使用Exists，检查某个或者某些key是否存在
	n,err:=rdb.Exists(ctx,"mykey").Result()
	if err!=nil{
		panic(err)
	}
	if n>0{
		fmt.Println("存在")
	}else{
		fmt.Println("不存在")
	}

	//使用Expire方法设置某个时间段（time.Duration）过期
	result3, err := rdb.Expire(ctx, "key", time.Minute*2).Result()
	if err!=nil{
		panic(err)
	}
	if result3{
		fmt.Println("设置成功")
	}else{
		fmt.Println("设置失败")
	}
	//使用ExpireAt方法设置在某个时间点（time.Time）过期
	result4, err := rdb.ExpireAt(ctx, "key2", time.Now()).Result()
	if err!=nil{
		panic(err)
	}
	if result4{
		fmt.Println("设置成功")
	}else{
		fmt.Println("设置失败")
	}

	//使用TTL()获取某个key的剩余有效期，单位是秒s
	ttl, err := rdb.TTL(ctx, "key").Result()
	if err!=nil{
		panic(err)
	}
	fmt.Println(ttl)
	//使用PTTL()获取某个key的剩余有效期，单位是毫秒ms
	pttl, err := rdb.PTTL(ctx, "key").Result()
	if err!=nil{
		panic(err)
	}
	fmt.Println(pttl)

	//使用DBSize()查看当前数据库的数量
	num, err := rdb.DBSize(ctx).Result()
	if err!=nil{
		panic(err)
	}
	fmt.Println("数据库的缓存项有：",num)

	//使用Del() 删除缓存项
	n, err = rdb.Del(ctx, "key", "key2").Result()
	if err!=nil{
		panic(err)
	}
	fmt.Println("成功删除了多少个：",n)

	//使用FlushDB() 清空当前数据
	result5, err := rdb.FlushDB(ctx).Result()
	if err!=nil{
		panic(err)
	}
	fmt.Println(result5)

	//使用FlushAll() 清空所有数据库
	result6, err := rdb.FlushAll(ctx).Result()
	if err!=nil{
		panic(err)
	}
	fmt.Println(result6)
}
/*执行结果：
Keys()结果：[lan_rank key book mykey key2]
Type()结果：zset
Exists()结果：存在
Expire()结果：设置成功
ExpireAt()结果：设置成功
TTL()结果：2m0s
PTTL()结果：1m59.98s
DBSize()结果：数据库的缓存项有： 4
Del()结果：成功删除了多少个： 1
FlushDB()结果：OK
FlushAll()结果：OK

*/

string类型

Set相关

Set方法的最后一个参数表示的过期时间，0表示永不过期 SetEx()方法：设置并指定过期时间 SetNx()方法：设置并制定过期时间，区别于SetEx在于，SetNx只能设置不存在的key

package main

import (
	"context"
	"fmt"
	"github.com/go-redis/redis/v8"
	"time"
)

var ctx=context.Background()

func main(){
	rdb:=redis.NewClient(&redis.Options{
		Addr:"192.168.145.151:6379",
		Password:"",
		DB:0,
	})

	//Set方法的最后一个参数表示的过期时间，0表示永不过期
	err	:= rdb.Set(ctx, "key1", "value1", 0).Err()
	if err!=nil{
		panic(err)
	}
	err = rdb.Set(ctx, "key2", "value2", time.Minute*2).Err()
	if err!=nil{
		panic(err)
	}

	//SetEx()方法：设置并指定过期时间
	err = rdb.SetEX(ctx, "key3", "value3", time.Hour*2).Err()
	if err!=nil{
		panic(err)
	}

	//SetNx()方法：设置并制定过期时间，区别于SetEx在于，SetNx只能设置不存在的key
	result, err := rdb.SetNX(ctx, "key3", "value3new", time.Minute).Result()
	if err!=nil{
		panic(err)
	}
	if result{
		fmt.Println("SetNx方法成功")
	}else{
		fmt.Println("key已经存在，SetNx方法设置失败")
	}
}

Get方法 获取key，如果key不存在，将返回redis.Nil GetRange:字符串截取

package main

import (
	"context"
	"fmt"
	"github.com/go-redis/redis/v8"
)

var ctx=context.Background()

func main(){
	rdb:=redis.NewClient(&redis.Options{
		Addr:"192.168.145.151:6379",
		Password:"",
		DB:0,
	})

	//Set方法的最后一个参数表示的过期时间，0表示永不过期
	err	:= rdb.Set(ctx, "key1", "value1", 0).Err()
	if err!=nil{
		panic(err)
	}

	//Get方法 获取key，如果key不存在，将返回redis.Nil
	result, err := rdb.Get(ctx, "key1").Result()
	if err!=nil{
		panic(err)
	}
	fmt.Println("Get方法获取key1的结果为：",result)
	//Get方法获取一个不存在的key，返回redis.Nil
	result2, err := rdb.Get(ctx, "no-key").Result()
	if err == redis.Nil{
		fmt.Println("key不存在")
	}else if err!=nil{
		panic(err)
	}else{
		fmt.Println("key的值为：",result2)
	}

	//GetRange:字符串截取
	err = rdb.Set(ctx, "mykey", "Hello redis",0).Err()
	if err!=nil{
		panic(err)
	}
	result3, err := rdb.GetRange(ctx, "mykey", 0, 5).Result()
	if err!=nil{
		panic(err)
	}
	fmt.Println("GetRange截取到的值为：",result3)
}

Get方法获取key1的结果为： value1
key不存在

Incr、Decr

Incr方法，对key的值进行加1 IncrBy()方法，按指定步长增加 Decr()；减1 Decrby()；按照指定步长减少

package main

import (
	"context"
	"fmt"
	"github.com/go-redis/redis/v8"
)

var ctx=context.Background()

func main(){
	rdb:=redis.NewClient(&redis.Options{
		Addr:"192.168.145.151:6379",
		Password:"",
		DB:0,
	})

	//Incr方法，对key的值进行加1
	result, err := rdb.Incr(ctx, "number").Result()
	if err!=nil{
		panic(err)
	}
	fmt.Println("key当前的值为：",result)

	//IncrBy()方法，按指定步长增加
	result2, err := rdb.IncrBy(ctx, "number", 5).Result()
	if err!=nil{
		panic(err)
	}
	fmt.Println("IncrBy后，当前的key的值为：",result2)

	//Decr()；减1
	result3, err := rdb.Decr(ctx, "number").Result()
	if err!=nil{
		panic(err)
	}
	fmt.Println("Decr后，当前的key的值为：",result3)

	//Decrby()；按照指定步长减少
	result4, err := rdb.DecrBy(ctx, "number", 2).Result()
	if err!=nil{
		panic(err)
	}
	fmt.Println("DecrBy后，当前的key的值为：",result4)
}
/* 执行结果：
key当前的值为： 1
IncrBy后，当前的key的值为： 6
Decr后，当前的key的值为： 5
DecrBy后，当前的key的值为： 3

list类型

LPush()，从添加数据到链表；返回值是当前列表元素的数量；也可以从右侧加入到链表，对应的方法是RPush() LInsert():在某个位置插入新元素 Lset()：设置某个元素的值；下标是从0开始的 LLen():获取链表元素个数 LIndex()：获取链表下标对应的元素 LPop（）：从链表左侧弹出数据；RPop()是从右侧弹出 LRem()：根据值移除元素

package main

import (
	"context"
	"fmt"
	"github.com/go-redis/redis/v8"
)

var ctx=context.Background()

func main(){
	rdb:=redis.NewClient(&redis.Options{
		Addr:"192.168.145.151:6379",
		Password:"",
		DB:0,
	})

	//LPush，从添加数据到链表；返回值是当前列表元素的数量
	//也可以从右侧加入到链表，对应的方法是RPush()
	n, err := rdb.LPush(ctx, "list", 1, 2, 3).Result()
	if err != nil {
		panic(err)
	}
	fmt.Println(n)

	//LInsert():在某个位置插入新元素
	//在名为key的缓存项值为100的元素前面插入一个值，值为123
	err = rdb.LInsert(ctx, "key", "before", "100", 123).Err()
	if err != nil {
		panic(err)
	}

	//Lset()：设置某个元素的值；下标是从0开始的
	err = rdb.LSet(ctx, "list", 1, 100).Err()
	if err != nil {
		panic(err)
	}

	//LLen():获取链表元素个数
	length, err := rdb.LLen(ctx, "list").Result()
	if err != nil {
		panic(err)
	}
	fmt.Printf("当前链表的长度为: %v\n", length)

	//LIndex()：获取链表下标对应的元素
	val, err := rdb.LIndex(ctx, "list", 0).Result()
	if err != nil {
		panic(err)
	}
	fmt.Printf("下标为0的值为: %v\n", val)

	//LRange()：获取某个选定范围的元素集
	vals, err := rdb.LRange(ctx, "list", 0, 2).Result()
	if err != nil {
		panic(err)
	}
	fmt.Printf("从下标0到下标2的值: %v\n", vals)

	//LPop（）：从链表左侧弹出数据；RPop()是从右侧弹出
	val, err = rdb.LPop(ctx, "list").Result()
	if err != nil {
		panic(err)
	}
	fmt.Printf("移除的元素为: %v\n", val)

	//LRem()：根据值移除元素
	num, err := rdb.LRem(ctx, "list", 2, "100").Result()
	if err != nil {
		panic(err)
	}
	fmt.Printf("移除了: %v 个\n", num)
}
/* 执行结果：
3
当前链表的长度为: 3
下标为0的值为: 3
从下标0到下标2的值: [3 100 1]
移除的元素为: 3
移除了: 1 个
*/

set类型

SAdd() 添加元素

SPop()：随机获取一个元素

SRem() ：删除集合里指定的值

SSMembers()：获取所有的成员

SIsMember()：判断元素是否在集合中

SCard()：获取集合个数

SUnion()：并集；SDiff():差集；SInter():交集

package main

import (
	"context"
	"fmt"
	"github.com/go-redis/redis/v8"
)

var ctx=context.Background()

func main(){
	rdb:=redis.NewClient(&redis.Options{
		Addr:"192.168.145.151:6379",
		Password:"",
		DB:0,
	})

	//SAdd() 添加元素
	rdb.SAdd(ctx, "lans", "go", "java")
	rdb.SAdd(ctx, "lans", "c++")
	rdb.SAdd(ctx, "lans", "go")

	//SPop()：随机获取一个元素
	val, err := rdb.SPop(ctx, "lans").Result()
	if err!=nil{
		panic(err)
	}
	fmt.Println(val)

	//SRem() ：删除集合里指定的值
	n, err := rdb.SRem(ctx, "lans", "java", "c++").Result()
	if err != nil {
		panic(err)
	}
	fmt.Println(n)

	//SSMembers()：获取所有的成员
	vals, err := rdb.SMembers(ctx, "lans").Result()
	if err != nil {
		panic(err)
	}
	fmt.Println(vals)

	//SIsMember()：判断元素是否在集合中
	exists, err := rdb.SIsMember(ctx, "lans", "java").Result()
	if err != nil {
		panic(err)
	}
	if exists {
		fmt.Println("存在集合中")
	} else {
		fmt.Println("不存在集合中")
	}

	//SCard()：获取集合个数
	total, err := rdb.SCard(ctx, "lans").Result()
	if err != nil {
		panic(err)
	}
	fmt.Printf("集合总共有 %v 个元素\n", total)

	//SUnion()：并集
	union, err := rdb.SUnion(ctx, "setA", "setB").Result()
	if err != nil {
		panic(err)
	}
	fmt.Println(union)

	rdb.SAdd(ctx, "setA", "a", "b", "c", "d")
	rdb.SAdd(ctx, "setB", "a", "d", "e", "f")

	//SDiff():差集
	diff, err := rdb.SDiff(ctx, "setA", "setB").Result()
	if err != nil {
		panic(err)
	}
	fmt.Println(diff)

	//SInter():交集
	inter, err := rdb.SInter(ctx, "setA", "setB").Result()
	if err != nil {
		panic(err)
	}
	fmt.Println(inter)
}

/* 执行结果：
java
1
[go]
不存在集合中
集合总共有 1 个元素
[c f b d e a]
[b c]
[d a]
*/

zset类型

ZAdd() 添加元素 ZIncrBy()：增加元素分值 ZRange() 根据Score升序排列 、ZRevRange() 根据Score降序排列 ZRangeByScore()：获取Score过滤后的升序排列； ZRevRangeByScore()：获取Score过滤后的降序排列； ZCard():获取元素个数 ZCount():获取区间内的元素个数 ZScore():获取元素的score ZRank():获取元素在集合中升序的排名，从0开始；ZRevRank()：降序的排名 ZRem()：通过元素的值来删除元素 ZRemRangeByRank():根据排名删除 ZRemRangeByScore():根据分值区间来删除

package main

import (
	"context"
	"fmt"
	"github.com/go-redis/redis/v8"
	"math/rand"
)

var ctx=context.Background()

func main(){
	rdb:=redis.NewClient(&redis.Options{
		Addr:"192.168.145.151:6379",
		Password:"",
		DB:0,
	})

	//ZAdd() 添加元素
	zSets:=[]*redis.Z{
		&redis.Z{Score:0,Member:1},
		&redis.Z{Score:0,Member:2},
		&redis.Z{Score:0,Member:3},
		&redis.Z{Score:0,Member:4},
		&redis.Z{Score:0,Member:5},
		&redis.Z{Score:0,Member:6},
	}
	rdb.ZAdd(ctx,"zSet",zSets...)

	//ZIncrBy()：增加元素分值
	rdb.ZIncrBy(ctx, "zSet", float64(rand.Intn(100)), "1")
	rdb.ZIncrBy(ctx, "zSet", float64(rand.Intn(100)), "2")
	rdb.ZIncrBy(ctx, "zSet", float64(rand.Intn(100)), "3")
	rdb.ZIncrBy(ctx, "zSet", float64(rand.Intn(100)), "4")
	rdb.ZIncrBy(ctx, "zSet", float64(rand.Intn(100)), "5")
	rdb.ZIncrBy(ctx, "zSet", float64(rand.Intn(100)), "6")

	//ZRange() 根据Score升序排列 、ZRevRange() 根据Score降序排列
	//获取分值(点击率)前三的文章ID列表
	res, err := rdb.ZRevRange(ctx, "zSet", 0, 2).Result()
	if err != nil {
		panic(err)
	}
	fmt.Println(res)

	//ZRangeByScore()：获取Score过滤后的升序排列； ZRevRangeByScore()：获取Score过滤后的降序排列；
	res2, err := rdb.ZRangeByScore(ctx, "zSet", &redis.ZRangeBy{
		Min:    "40",
		Max:    "85",
	}).Result()

	if err != nil {
		panic(err)
	}
	fmt.Println(res2)

	//ZCard():获取元素个数
	count, err := rdb.ZCard(ctx, "zSet").Result()
	if err != nil {
		panic(err)
	}
	fmt.Println(count)

	//ZCount():获取区间内的元素个数
	n, err := rdb.ZCount(ctx, "zSet", "40", "85").Result()
	if err != nil {
		panic(err)
	}
	fmt.Println(n)

	//ZScore():获取元素的score
	score, err := rdb.ZScore(ctx, "zSet", "5").Result()
	if err != nil {
		panic(err)
	}
	fmt.Println(score)

	//ZRank():获取元素在集合中升序的排名，从0开始；ZRevRank()：降序的排名
	res3, err := rdb.ZRevRank(ctx, "zSet", "2").Result()
	if err != nil {
		panic(err)
	}
	fmt.Println("ZRevRank方式排名：",res3)

	//ZRem()：通过元素的值来删除元素
	res4, err := rdb.ZRem(ctx, "zSet", "2").Result()
	if err != nil {
		panic(err)
	}
	fmt.Println(res4)

	//ZRemRangeByRank():根据排名删除
	//按照升序排序删除第一个和第二个元素
	res5, err := rdb.ZRemRangeByRank(ctx, "zSet",  0, 1).Result()
	if err != nil {
		panic(err)
	}
	fmt.Println(res5)

	//ZRemRangeByScore():根据分值区间来删除
	//删除score在[40, 70]之间的元素
	res6, err := rdb.ZRemRangeByScore(ctx, "zSet", "40", "70").Result()
	if err != nil {
		panic(err)
	}
	fmt.Println(res6)
}

hash类型

HSet():设置 HMset():批量设置 HGet():获取某个元素 HGetAll():获取全部元素 HDel():删除某个元素 HExists():判断元素是否存在 HLen():获取长度

package main

import (
	"context"
	"fmt"
	"github.com/go-redis/redis/v8"
)

var ctx=context.Background()

func main(){
	rdb:=redis.NewClient(&redis.Options{
		Addr:"192.168.145.151:6379",
		Password:"",
		DB:0,
	})

	//HSet():设置
	rdb.HSet(ctx, "user", "key1", "value1", "key2", "value2")
	rdb.HSet(ctx, "user", []string{"key3", "value3", "key4", "value4"})
	rdb.HSet(ctx, "user", map[string]interface{}{"key5": "value5", "key6": "value6"})

	//HMset():批量设置
	rdb.Del(ctx, "user")
	rdb.HMSet(ctx, "user", map[string]interface{}{"name":"abin", "age": 18, "address":"中国"})

	//HGet():获取某个元素
	address, err := rdb.HGet(ctx, "user", "address").Result()
	if err != nil {
		panic(err)
	}
	fmt.Println(address)

	//HGetAll():获取全部元素
	user, err := rdb.HGetAll(ctx, "user").Result()
	if err != nil {
		panic(err)
	}
	fmt.Println(user)

	//HDel():删除某个元素
	res, err := rdb.HDel(ctx, "user", "name", "age").Result()
	if err != nil {
		panic(err)
	}
	fmt.Println(res)

	//HExists():判断元素是否存在
	res2, err := rdb.HExists(ctx, "user", "address").Result()
	if err != nil {
		panic(err)
	}
	fmt.Println(res2)

	//HLen():获取长度
	res3, err := rdb.HLen(ctx, "user").Result()
	if err != nil {
		panic(err)
	}
	fmt.Println(res3)
}
/* 执行结果：
中国
map[address:中国 age:18 name:abin]
2
true
1
*/
