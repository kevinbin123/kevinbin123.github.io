Golang小案例-爬取豆瓣电影

在之前有学习过其他的爬虫案例，这个要实现的这个案例不一样的地方在于用到了第三方选择器、以及持久化存储是使用数据库

案例来源：

小生凡一的博客：Go语言并发爬虫

https://blog.csdn.net/weixin_45304503/article/details/121885673

功能

1、通过url访问豆瓣的网址

2、通过第三方解析器解析内容；

3、持久化到本地数据库中

实现

1、定义要保存的电影的数据结构


type MovieData struct {
	Title    string `json:"title"`
	Director string `json:"director"`
	Picture  string `json:"picture"`
	Actor    string `json:"actor"`
	Year     string `json:"year"`
	Score    string `json:"score"`
	Quote    string `json:"quote"`
}

2、初始化数据库

var DB *sql.DB
const (
	USERNAME = "root"
	PASSWORD = "123456"
	HOST     = "localhost"
	PORT     = "3306"
	DBNAME   = "douban_movie"
)

func InitDB() {
	//数据库连接格式：root:123456@tcp(127.0.0.1:3306)/douban_movie?charset=utf8
	path := strings.Join([]string{USERNAME, ":", PASSWORD, "@tcp(", HOST, ":", PORT, ")/", DBNAME, "?charset=utf8"}, "")
	DB, _ = sql.Open("mysql", path)
	DB.SetConnMaxLifetime(10)
	DB.SetMaxIdleConns(5)
	if err := DB.Ping(); err != nil {
		fmt.Println("open database fail", err)
		return
	}
	fmt.Println("connect database success")
}

3、准备好数据库表

create database douban_movie;
use douban_movie;
create table movie_data(
	id int primary key auto_increment;
	Title varchar(255),
	Director varchar(255),
	Picture varchar(255),
	Actor varchar(255),
	Year varchar(255),
	Score varchar(255),
 	Quote varchar(255)
);

4、开始爬取网页

func Spider(url string) {
	fmt.Println("当前爬取：", url)
	client := &http.Client{}
	req, err := http.NewRequest("GET", url, nil)
	if err != nil {
		log.Fatal(err)
	}
	req.Header.Set("Connection", "keep-alive")
	req.Header.Set("Cache-Control", "max-age=0")
	req.Header.Set("sec-ch-ua-mobile", "?0")
	req.Header.Set("Upgrade-Insecure-Requests", "1")
	req.Header.Set("User-Agent", "Mozilla/5.0 (Windows NT 10.0; Win64; x64) AppleWebKit/537.36 (KHTML, like Gecko) Chrome/96.0.4664.110 Safari/537.36")
	req.Header.Set("Accept", "text/html,application/xhtml+xml,application/xml;q=0.9,image/avif,image/webp,image/apng,*/*;q=0.8,application/signed-exchange;v=b3;q=0.9")
	req.Header.Set("Sec-Fetch-Site", "same-origin")
	req.Header.Set("Sec-Fetch-Mode", "navigate")
	req.Header.Set("Sec-Fetch-User", "?1")
	req.Header.Set("Sec-Fetch-Dest", "document")
	req.Header.Set("Referer", "https://movie.douban.com/chart")
	req.Header.Set("Accept-Language", "zh-CN,zh;q=0.9")
	resp, err := client.Do(req)
	if err != nil {
		log.Fatal(err)
	}
	defer resp.Body.Close()
	docDetail, err := goquery.NewDocumentFromReader(resp.Body)
	if err != nil {
		log.Fatal(err)
	}
	docDetail.Find("#content > div > div.article > ol > li > div").
		Each(func(i int, s *goquery.Selection) {
			var md MovieData
			title := s.Find("div.info > div.hd > a > span:nth-child(1)").Text()
			//fmt.Println("title:", title)
			imgNode := s.Find("div.pic > a > img")
			img, ok := imgNode.Attr("src")
			info := strings.Trim(s.Find("div.info > div.bd > p:nth-child(1)").Text(), " ")
			director, actor, year := InfoSplit(info)
			score := strings.Trim(s.Find("div.info > div.bd > div > span.rating_num").Text(), " ")
			score = strings.Trim(score, "\n")
			quote := strings.Trim(s.Find("div.info > div.bd > p.quote > span").Text(), " ")
			if ok {
				md.Title = title
				md.Picture = img
				md.Director = director
				md.Actor = actor
				md.Year = year
				md.Score = score
				md.Quote = quote
				if !InsertSql(md) {
					fmt.Println("插入失败")
				} else {
					fmt.Println(title, "插入成功")
				}
			}
		})
}
func InfoSplit(info string) (director, actor, year string) {
	//导演: 弗兰克·德拉邦特 Frank Darabont   主演: 蒂姆·罗宾斯 Tim Robbins /...
	//1994 / 美国 / 犯罪 剧情
	directorRe, _ := regexp.Compile(`导演: (.*)主演:`)
	director = strings.Trim(string(directorRe.Find([]byte(info))), "主演:")
	actorRe, _ := regexp.Compile(`主演: (.*)`)
	actor = string(actorRe.Find([]byte(info)))
	yearRe, _ := regexp.Compile(`(\d+)`)
	year = string(yearRe.Find([]byte(info)))
	return

}

5、插入数据

func InsertSql(md MovieData) bool {
	/**
	fmt.Println("md:Title", md.Title)
	fmt.Println("md:Director", md.Director)
	fmt.Println("md:Picture", md.Picture)
	fmt.Println("md:Actor", md.Actor)
	fmt.Println("md:Year", md.Year)
	fmt.Println("md:Score", md.Score)
	fmt.Println("md:Quote", md.Quote)
	*/

	tx, err := DB.Begin()
	if err != nil {
		fmt.Println("tx fail", err)
		return false
	}
	stmt, err := tx.Prepare("insert into movie_data(`Title`,`Director`,`Picture`,`Actor`,`Year`,`Score`,`Quote`) values(?,?,?,?,?,?,?)")
	if err != nil {
		fmt.Println("prepare fail", err)
		return false
	}
	_, err = stmt.Exec(md.Title, md.Director, md.Picture, md.Actor, md.Year, md.Score, md.Quote)
	if err != nil {
		fmt.Println("Exec fail", err)
		return false
	}
	err = tx.Commit()
	if err != nil {
		fmt.Println("commit fail", err)
		return false
	}
	return true
}

6、开启爬取工作

这边，启用了goroutine去爬取，只爬取了4页，太多了，怕被封。

func main() {
	InitDB()
	wg := sync.WaitGroup{}
	wg.Add(4)
	for i := 0; i < 4; i++ {
		go func(i int) {
			defer wg.Done()
			Spider("https://movie.douban.com/top250?start=" + strconv.Itoa(i*25))
		}(i)
	}
	wg.Wait()
}
