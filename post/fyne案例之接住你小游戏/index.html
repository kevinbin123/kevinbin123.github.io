​ 1、自定义主题

​ 自定义一个myTheme.go的文件，然后用到了是微软雅黑的字体。

package main

import (
	_ "embed"
	"fyne.io/fyne/v2"
	"image/color"

	"fyne.io/fyne/v2/theme"
)

//go:embed msyh.ttc
var font []byte

var myfont = &fyne.StaticResource{
	StaticName:    "MSYH",
	StaticContent: font,
}

type myTheme struct{}

var _ fyne.Theme = (*myTheme)(nil)

// return bundled font resource
func (*myTheme) Font(s fyne.TextStyle) fyne.Resource {
	if s.Monospace {
		return theme.LightTheme().Font(s)
		// return theme.DefaultTheme().Font(s)
	}
	if s.Bold {
		if s.Italic {
			return theme.LightTheme().Font(s)
			// return theme.DefaultTheme().Font(s)
		}
		return myfont
	}
	if s.Italic {
		return theme.LightTheme().Font(s)
		// return theme.DefaultTheme().Font(s)
	}
	return myfont
}

func (*myTheme) Color(n fyne.ThemeColorName, v fyne.ThemeVariant) color.Color {
	// 自定义背景
	if n == theme.ColorNameBackground {
		return hex2rgb("dddddd")
	}
	if n == theme.ColorNameForeground {
		return hex2rgb("ff3300")
	}
	if n == theme.ColorNameDisabled {
		return hex2rgb("ffffff")
	}
	if n == theme.ColorNameDisabledButton {
		return hex2rgb("550033")
	}
	return theme.LightTheme().Color(n, v)
}

func (*myTheme) Icon(n fyne.ThemeIconName) fyne.Resource {
	return theme.LightTheme().Icon(n)
	// return theme.DefaultTheme().Icon(n)
}

func (*myTheme) Size(n fyne.ThemeSizeName) float32 {
	return theme.LightTheme().Size(n)
	// return theme.DefaultTheme().Size(n)
}

​ 2、先搭建游戏的框架界面，并设置自定义的主题。

func main() {
	ap := app.New()
	w := ap.NewWindow("Get U")
	ap.Settings().SetTheme(&myTheme{})
	w.Resize(fyne.NewSize(630, 630))
	bg := canvas.NewRectangle(hex2rgb("ffffff"))
	
	bg.Resize(fyne.NewSize(600, 600))
	bg.StrokeColor = hex2rgb("ff0000")
	bg.StrokeWidth = 5

	cc := container.NewWithoutLayout(bg)
	c := container.NewBorder(nil, nil, nil, nil, cc)
	w.SetContent(c)
	w.CenterOnScreen()
	w.SetMaster()
	w.ShowAndRun()

}
func hex2rgb(str string) color.Color {
	if len(str) != 6 {
		return color.Black
	}
	r, _ := strconv.ParseInt(str[:2], 16, 10)
	g, _ := strconv.ParseInt(str[2:4], 16, 18)
	b, _ := strconv.ParseInt(str[4:], 16, 10)
	return color.RGBA{uint8(r), uint8(g), uint8(b), 255}
}

	fi, err := ioutil.ReadDir(pth)

​ 5、设置开始，以及分数

func main() {
	ap := app.New()
	w := ap.NewWindow("Get U")
	ap.Settings().SetTheme(&myTheme{})
	w.Resize(fyne.NewSize(630, 630))
	bg := canvas.NewRectangle(hex2rgb("ffffff"))
	//计数器
	//lblResult := widget.NewLabel("total point")
	lblResult := canvas.NewText("", hex2rgb("0000ff"))
	lblResult.Text = "0"
	lblResult.TextSize = 50
	result := 0
	//下落物
	foods := GetFileList("img")
	food := canvas.NewImageFromFile("img/banana.png")
	food1 := canvas.NewImageFromFile("img/caomei.png")
	foodSize := fyne.NewSize(50, 50)
	food.Resize(foodSize)
	food1.Resize(foodSize)
	food.Move(fyne.NewPos(100, 50))
	food1.Move(fyne.NewPos(200, 50))
	//circle := canvas.NewCircle(hex2rgb("ff0000"))
	//circle.Resize(fyne.NewSize(50, 50))
	//circle.Move(fyne.NewPos(100, 50))

	//小车
	rect := canvas.NewRectangle(hex2rgb("ffffff"))
	rect.Resize(fyne.NewSize(100, 50))
	rect.Move(fyne.NewPos(100, 540))
	rect.StrokeWidth = 5
	rect.StrokeColor = hex2rgb("0000ff")

	//下落物移动
	t := time.NewTicker(time.Millisecond * 100)
	rand.Seed(time.Now().UnixNano())
    //重置下落物，包括重置物品、以及开始下落的位置
	foodReset := func(food *canvas.Image) {
		res, _ := ioutil.ReadFile(foods[rand.Intn(len(foods))])
		food.Resource = fyne.NewStaticResource("img", res)
		food.Move(fyne.NewPos(rand.Float32()*550, 50))
	}
	//定义速度和分数初始
	var startSpeed float32 = 0
	var speeds = []Speed{&uniform{}, &double{}, &exponent{}, &dumps{}, &sqrt{}}
	//
	foodRefresh := func(food *canvas.Image) {
        //随机选择下落速度的模式，有匀速下落、两倍速、指数、减速、开更号等
		randSpeed := speeds[rand.Intn(len(speeds))]
		for range t.C {
            //如果下落物已经落到了地面，则重置下落物
			if food.Position().Y > 600 {
				foodReset(food)
				startSpeed = 0
				randSpeed = speeds[rand.Intn(len(speeds))]
			}
            // 下落物的速度逐渐加快
			startSpeed += 0.5
			food.Move(food.Position().Add(fyne.NewDelta(0, randSpeed.speed(startSpeed))))
            //判断下落物是否已经落进了小车中
			if food.Position().X > rect.Position().X && food.Position().X < rect.Position().X+100 && food.Position().Y > rect.Position().Y {
				foodReset(food)
				startSpeed = 0
				randSpeed = speeds[rand.Intn(len(speeds))]
				result += randSpeed.point()
				lblResult.Text = strconv.Itoa(result)
				lblResult.Refresh()
			}
			food.Refresh()
		}
	}
	btnStart := widget.NewButton("开始", func() {
		go func() {
			foodRefresh(food)
		}()
		go func() {
			foodRefresh(food1)
		}()
	})

	//小车移动
	w.Canvas().SetOnTypedKey(func(event *fyne.KeyEvent) {
		switch event.Name {
		case fyne.KeyLeft:
			if rect.Position().X > 0 {
				rect.Move(rect.Position().Add(fyne.NewDelta(-10, 0)))
			}
		case fyne.KeyRight:
			if rect.Position().X < 500 {
				rect.Move(rect.Position().Add(fyne.NewDelta(10, 0)))
			}
		}

	})
	//设置背景框
	bg.Resize(fyne.NewSize(600, 600))
	bg.StrokeColor = hex2rgb("ff0000")
	bg.StrokeWidth = 5
	//组装各个布局
    //右侧由背景、下落物、小车组成
	cc := container.NewWithoutLayout(bg, food, food1, rect)
    //左侧由计分器、开始按钮组成
	left := container.NewVBox(lblResult, btnStart)
	c := container.NewBorder(nil, nil, left, nil, cc)
	w.SetContent(c)
	w.CenterOnScreen()
	w.SetMaster()
	w.ShowAndRun()

}
