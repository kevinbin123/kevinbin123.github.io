func encoding(r *http.Response) {

package main

import (
	"fmt"
	"io"
	"net/http"
	"os"
)


type Reader struct {
	io.Reader
	Total   int64
	Current int64
}

func downloadFileProgress(url, filename string) {
	resp, err := http.Get(url)
	if err != nil {
		panic(err)
	}
	defer resp.Body.Close()
	file, err := os.Create(filename)
	if err != nil {
		panic(err)
	}
	defer file.Close()
	reader := &Reader{
		Reader: resp.Body,
		Total:  resp.ContentLength,
	}
	io.Copy(file, reader)
}

func (r *Reader) Read(p []byte) (n int, e error) {
	n, e = r.Reader.Read(p)
	r.Current += int64(n)
	fmt.Printf("\r进度%.2f%%", float64(r.Current*10000/r.Total)/100)
	return
}

func rrcookie() {
	client := &http.Client{
		CheckRedirect: func(req *http.Request, via []*http.Request) error {
			return http.ErrUseLastResponse
		},
	}
	firstRequest, _ := http.NewRequest(http.MethodGet, "http://httpbin.org/cookies/set?name=abin&password=123456", nil)
	firstResp, err := client.Do(firstRequest)
	if err != nil {
		panic(err)
	}
	defer firstResp.Body.Close()

	secondRequest, _ := http.NewRequest(http.MethodGet, "http://httpbin.org/cookies", nil)
	for _, cookie := range firstResp.Cookies() {
		secondRequest.AddCookie(cookie)
	}

	secondResp, err := client.Do(secondRequest)
	if err != nil {
		panic(err)
	}
	defer secondResp.Body.Close()
	content, _ := ioutil.ReadAll(secondResp.Body)
	fmt.Printf("%s", content)
}

go get  github.com/juju/persistent-cookiejar

创建一个server，提供login和center两个路由；

1、先通过login登录，再访问center

2、携带了cookie就返回“个人主页”；没有携带cookie就返回”无法访问“

package main

import (
	"net/http"
	"time"
)

func login(w http.ResponseWriter, req *http.Request) {
	username := req.PostFormValue("username")
	password := req.PostFormValue("password")
	if username == "abin" && password == "abin123" {
		http.SetCookie(w, &http.Cookie{
			Name:    "isLogin",
			Value:   "1",
			Expires: time.Now().Add(3 * time.Hour),
		})
		_, _ = w.Write([]byte("登录成功\n"))
	} else {
		_, _ = w.Write([]byte("登录失败"))
	}
	return
}
func center(w http.ResponseWriter, r *http.Request) {
	isLogin, err := r.Cookie("isLogin")
	if err == http.ErrNoCookie {
		_, _ = w.Write([]byte("无法访问"))
		w.WriteHeader(http.StatusUnauthorized)
		return
	}
	if isLogin.Value != "1" {
		_, _ = w.Write([]byte("无法访问"))
		w.WriteHeader(http.StatusUnauthorized)
		return
	}
	_, _ = w.Write([]byte("个人主页\n"))
}

func main() {
	http.HandleFunc("/login", login)
	http.HandleFunc("/center", center)
	_ = http.ListenAndServe(":8080", nil)
}

package main

	"net/http/cookiejar"
	"net/url"

)

func login(jar http.CookieJar) {
	client := &http.Client{
		Jar: jar,
	}
	r, _ := client.PostForm("http://localhost:8080/login",
		url.Values{"username": {"abin"}, "password": {"abin123"}})
	defer r.Body.Close()
	_, _ = io.Copy(os.Stdout, r.Body)
}

func center(jar http.CookieJar) {
	client := &http.Client{
		Jar: jar,
	}
	r, _ := client.Get("http://localhost:8080/center")
	defer r.Body.Close()
	_, _ = io.Copy(os.Stdout, r.Body)
}

func main() {
	// 会话期cookie

	// 持久性 cookie  第三方 github.com/juju/persistent-cookiejar
	jar, _ := cookiejar2.New(nil)
	//login(jar)
	center(jar)
	jar.Save()

	// 查看cookie存放位置 cat ~/.go-cookies

func main() {
	http.HandleFunc("/process", func(w http.ResponseWriter, r *http.Request) {
		switch r.Method {
		case http.MethodPost:
			dec := json.NewDecoder(r.Body)
			com := company{}
			err := dec.Decode(&com)
			if err != nil {
				log.Println(err.Error())
				w.WriteHeader(http.StatusInternalServerError)
				return
			}
			enc := json.NewEncoder(w)
			err = enc.Encode(com)
			if err != nil {
				log.Println(err.Error())
				w.WriteHeader(http.StatusInternalServerError)
				return
			}
		default:
			w.WriteHeader(http.StatusMethodNotAllowed)
		}

	})
	http.ListenAndServe(":8080", nil)
}

type AuthMiddle struct {
	Next http.Handler
}

func (a *AuthMiddle) ServeHTTP(w http.ResponseWriter, r *http.Request) {
	if a.Next == nil {
		a.Next = http.DefaultServeMux
	}
	if r.Header.Get("Authrization") != "" {
		a.Next.ServeHTTP(w, r)
	} else {
		w.WriteHeader(http.StatusUnauthorized)
	}
}

type TimeoutMiddle struct {
	Next http.Handler
}

func (tm *TimeoutMiddle) ServeHTTP(w http.ResponseWriter, r *http.Request) {
	if tm.Next == nil {
		tm.Next = http.DefaultServeMux
	}
	ctx := r.Context()
	ctx, _ = context.WithTimeout(ctx, 3*time.Second)
	r.WithContext(ctx)
	ch := make(chan struct{})
	go func() {
		tm.Next.ServeHTTP(w, r)
		ch <- struct{}{}
	}()
	select {
	case <-ch:
		return
	case <-ctx.Done():
		w.WriteHeader(http.StatusRequestTimeout)
	}
	ctx.Done()
}
